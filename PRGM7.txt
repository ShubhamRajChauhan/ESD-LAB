    PRESERVE8
    THUMB
    AREA RESET, CODE, READONLY
    EXPORT _Vectors
_Vectors
    DCD 0X20001000
    DCD START
    ALIGN
    AREA MY_CODE, CODE, READONLY
ENTRY
    EXPORT START
NUMBERS DCD 0xEDCBA988, 0x72B76799, 0xCCCCCCCC, 0xFFFFFFFF
COUNT DCD 4
NEGATIVE_COUNT DCD 0
START
    LDR R0, =NUMBERS
    LDR R1, =COUNT
    LDR R1,[R1]
    MOVS R2,#0
SCAN_LOOP
    LDR R3,[R0]
    LSRS R4,R3,#31
    ADDS R2,R2,R4
    ADDS R0,R0,#4
    SUBS R1,R1,#1
    BNE SCAN_LOOP
    LDR R0, =NEGATIVE_COUNT
    STR R2,[R0]
STOP B STOP
END
